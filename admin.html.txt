<script>
  const API = 'https://temple-queue-prediction.onrender.com';
  let PASSWORD    = '';
  let crowdCount  = 0;
  let servedCount = 0;

  async function login() {
    const pwd   = document.getElementById('password').value;
    const errEl = document.getElementById('login-err');
    errEl.classList.remove('show');
    if (!pwd) { errEl.textContent = '⚠️ Please enter password.'; errEl.classList.add('show'); return; }
    try {
      const res  = await fetch(API + '/admin/login', {
        method:'POST', headers:{'Content-Type':'application/json'},
        body: JSON.stringify({ password: pwd })
      });
      const data = await res.json();
      if (data.status === 'success') {
        PASSWORD = pwd;
        document.getElementById('login-section').style.display = 'none';
        document.getElementById('admin-section').style.display = 'block';
        loadQueue();
        setInterval(loadQueue, 10000);
      } else {
        errEl.textContent = '⚠️ Wrong password. Try again.';
        errEl.classList.add('show');
      }
    } catch {
      errEl.textContent = '⚠️ Cannot connect to server.';
      errEl.classList.add('show');
    }
  }

  async function callNext() {
    const btn = document.getElementById('call-btn');
    btn.classList.add('loading'); btn.disabled = true;
    try {
      const res  = await fetch(API + '/admin/call-next', {
        method:'POST', headers:{'Content-Type':'application/json'},
        body: JSON.stringify({ password: PASSWORD })
      });
      const data = await res.json();
      if (data.status === 'success') {
        document.getElementById('ct-empty').style.display = 'none';
        document.getElementById('ct-info').style.display  = 'block';
        document.getElementById('ct-token').textContent   = 'T-' + data.called_token;
        document.getElementById('ct-name').textContent    = data.called_name;
        servedCount++;
        document.getElementById('stat-served').textContent = servedCount;
        loadQueue();
      } else { alert(data.error || 'No one in queue!'); }
    } catch { alert('Cannot connect to server.'); }
    finally { btn.classList.remove('loading'); btn.disabled = false; }
  }

  async function markDone(token) {
    try {
      const res  = await fetch(API + '/admin/mark-done', {
        method:'POST', headers:{'Content-Type':'application/json'},
        body: JSON.stringify({ password: PASSWORD, token: token })
      });
      const data = await res.json();
      if (data.status === 'success') { servedCount++; loadQueue(); }
      else alert(data.error);
    } catch { alert('Cannot connect to server.'); }
  }

  async function loadQueue() {
    try {
      const res  = await fetch(API + '/queue/status');
      const data = await res.json();
      document.getElementById('stat-waiting').textContent = data.total_waiting;
      document.getElementById('stat-served').textContent  = servedCount;

      const crowd = await fetch(API + '/crowd/live');
      const cd    = await crowd.json();
      document.getElementById('stat-crowd').textContent = cd.count;
      crowdCount = cd.count;
      document.getElementById('crowd-num').textContent   = crowdCount;

      const container = document.getElementById('queue-items');
      if (data.total_waiting === 0) {
        container.innerHTML = '<div class="empty-q">🙏 No one is waiting right now</div>';
      } else {
        container.innerHTML = data.queue.map(p => `
          <div class="q-item">
            <div class="q-info">
              <span class="q-token">T-${p.token}</span>
              <div>
                <div class="q-name">${p.name}</div>
                <div class="q-time">Joined at ${p.joined_at}</div>
              </div>
            </div>
            <button class="done-btn" onclick="markDone(${p.token})">✅ Done</button>
          </div>
        `).join('');
      }
    } catch { console.log('Error loading queue'); }
  }

  function changeCrowd(delta) {
    crowdCount = Math.max(0, crowdCount + delta);
    document.getElementById('crowd-num').textContent = crowdCount;
  }

  async function sendCrowd() {
    try {
      const res  = await fetch(API + '/admin/crowd', {
        method:'POST', headers:{'Content-Type':'application/json'},
        body: JSON.stringify({ password: PASSWORD, count: crowdCount })
      });
      const data = await res.json();
      if (data.status === 'success') {
        document.getElementById('stat-crowd').textContent = crowdCount;
        alert('✅ Crowd count updated to ' + crowdCount);
      }
    } catch { alert('Cannot connect to server.'); }
  }

  async function resetQueue() {
    if (!confirm('Are you sure you want to reset the entire queue?')) return;
    try {
      const res  = await fetch(API + '/admin/reset', {
        method:'POST', headers:{'Content-Type':'application/json'},
        body: JSON.stringify({ password: PASSWORD })
      });
      const data = await res.json();
      if (data.status === 'success') {
        servedCount = 0;
        document.getElementById('ct-empty').style.display = 'block';
        document.getElementById('ct-info').style.display  = 'none';
        loadQueue();
        alert('✅ Queue reset successfully!');
      }
    } catch { alert('Cannot connect to server.'); }
  }
</script>
